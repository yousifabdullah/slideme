16      private final PriorityQueue <State> queue = new PriorityQueue<>(100, (State a, State b) -> a.priority() - b.priority());
17      
18      private final Board board;
19      
20      /**
21       * Alustaa tekoälyn nykyisen pelitilanteen mukaisesti.
22       * 
23       * @param board nykyinen pelitilanne Board-tietueena
24       */
25      public Solver(Board board) {
26          this.board = board;
27      }
28      
29      public int[] findPath() {
30          
31          queue.clear();
32          boolean[] closed = new boolean[876543210];
33          
34          queue.add(new State(this.board.getCurrentState()));
35          
36          while (!queue.isEmpty()) {
37              
38              State state = queue.poll();
39              
40              if (Array.matches(state.getCurrentIteration(), this.board.getFinalState())) {
41                  return this.tracePath(state);
42              }
43              
44              closed[this.getReference(state)] = true;
45              
46              for (int i = 0; i < 4; i++) {
47                  State successor = state.nextState(i);
48                  
49                  if (successor != null && closed[this.getReference(successor)] == false) {
50                      queue.add(successor);
51                  }
52              }
53          }
54          
55          return null;
56      }
57      
58      int getReference(State state) {
59          return Array.asInteger(state.getCurrentIteration()) - 1;
60      }
61      
62      /*
63      Lasketaan Manhattan-etäisyys indeksien a ja b välillä oleville
64      peliruuduille.
65      */
66      static int manhattanDistance(int a, int b) {
67          return Math.abs(a / 3 - b / 3) + Math.abs(a % 3 - b % 3);
68      }
69      
70      /*
71      A*-algoritmin heuristiikkana käytetään maksimia kaikkien peliruutujen
72      Manhattan-etäisyydestä.
73      */
74      static int heuristic(int[] tiles) {
75          int heuristic = 0;
76          
77          for (int i = 0; i < tiles.length; i++) {
78              // Huom. 0 merkitsee vapaaruutua, ei lukua 0!
79              if (tiles[i] != 0) {
80                  heuristic = Math.max(heuristic, manhattanDistance(i, tiles[i]));
81              }
82          }
83          
84          return heuristic;
85      }
