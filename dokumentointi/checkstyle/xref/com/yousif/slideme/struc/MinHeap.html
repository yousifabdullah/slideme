3   /**
4    * Oma toteutus minimikeosta, joka korvaa PriorityQueue-tietorakenteen.
5    * 
6    * @author Yousif Abdullah {@literal<yousif.abdullah@helsinki.fi>}
7    * @param <State> State-tietueet vertailtavina solmuina
8    */
9   public final class MinHeap<State extends Comparable<State>> {
10      
11      private State[] heap;
12      private int size;
13      
14      /**
15       * Alustaa minimikeon tyhjällä taulukolla. Oletusarvoisesti minimikeon
16       * taulukon koko on 2:n alkion suuruinen, joka voidaan tarpeen mukaan
17       * kasvattaa resize()-metodin avulla aina kaksinkertaiseksi.
18       * 
19       * @see com.yousif.slideme.struc.MinHeap#resize()
20       */
21      public MinHeap() {
22          this.heap = (State[]) new Comparable[2];
23          this.size = 0;
24      }
25      
26      /**
27       * Rakentaa minimikeon kekoehtoa noudattaen. Minimikeko rakennetaan eli
28       * korjataan kekoehtoa noudattaen bubbleDown()-metodin avulla.
29       * 
30       * Vaikka algoritmi kutsuu bubbleDown()-metodia n / 2 kertaa eli läpi-
31       * käytävien solmujen verran, on aikavaativuus pahimmassa tapauksessa
32       * O(n), siitä huolimatta, että bubbleDown()-metodin oma aikavaativuus on
33       * O(log n). Tarvittavien askeleiden määrä n-kokoisen keon korjaamiseen
34       * voidaan esittää matemaattisesti N = summa kaikista i = 0 käy log(n),
35       * jossa laskukaava on (n / 2 ^ (i + 1) * i). Koska algoritmi käy vain
36       * n / 2 solmua läpi, voidaan yleisemmin muotoilla (n / 2) * 2 = n ja
37       * n >= N, siis aikavaativuus on myös O(n) eikä O(n log n).
38       * 
39       * @see com.yousif.slideme.struc.MinHeap#bubbleDown(int)
40       */
41      private void construct() {
42          for (int i = this.size / 2; i > 0; i--) {
43              this.bubbleDown(i);
44          }
45      }
46      
47      /**
48       * Suorittaa "heapify"-toiminnon kekoehdon täyttämiseksi annetusta
49       * solmusta alaspäin. Toiminto siirtää käsiteltävän solmun omalle
50       * paikalleen minimikeossa sen prioriteetin perusteella ja jatkaa
51       * toimintaansa rekursiivisesti, kunnes kaikki solmut annetusta solmun
52       * indeksistä täyttävät kekoehdon. Toiminnon aikavaativuus on O(log n).
53       * 
54       * @param index käsiteltävän solmun indeksi minimikeon taulukossa
55       */
56      private void bubbleDown(int index) {
57          State current = this.heap[index];
58          int child;
59          
60          while (index * 2 <= this.size) {
61              child = index * 2;
62              
63              if (child != this.size) {
64                  if (this.heap[child].compareTo(this.heap[child + 1]) > 0) {
65                      child++;
66                  }
67              }
68              
69              if (current.compareTo(this.heap[child]) > 0) {
70                  this.heap[index] = this.heap[child];
71              } else {
72                  break;
73              }
74              
75              index = child;
76          }
77          
78          this.heap[index] = current;
79      }
80      
81      /**
82       * Lisää annetun iteraation minimikekoon sen prioriteetin perusteella.
83       * Toiminnon aikavaativuus on O(log n).
84       * 
85       * @param state lisättävä State-tietue
86       */
87      public void insert(State state) {
88          // Mikäli minimikeon taulukko on liian pieni, kasvatetaan taulukkoa.
89          if (this.size == this.heap.length - 1) {
90              this.resize();
91          }
92          
93          this.size++;
94          int index = this.size;
95          
96          while (index > 1 && state.compareTo(this.heap[index / 2]) < 0) {
97              this.heap[index] = this.heap[index / 2];
98              index = index / 2;
99          }
100         
101         this.heap[index] = state;
102     }
103     
104     /**
105      * Palauttaa sekä poistaa pienimmän prioriteetin solmun minimikeossa.
106      * Toiminnon aikavaativuus on O(log n).
107      * 
108      * @return pienimmän prioriteetin State-tietue
109      */
110     public State retrieve() {
111         if (this.size != 0) {
112             State state = this.heap[1];
113             
114             this.heap[1] = this.heap[this.size];
115             this.size--;
116             
117             this.bubbleDown(1);
118             return state;
119         }
120         
121         // Mikäli minimikeko on tyhjä, palautetaan null.
122         return null;
123     }
124     
125     /**
126      * Tarkistaa, onko minimikeko tyhjä. Toiminnon aikavaativuus on O(1).
127      * 
128      * @return true, kun minimikeko on tyhjä ja muutoin false
129      */
130     public boolean isEmpty() {
131         return (this.size == 0);
132     }
133     
134     /**
135      * Kasvattaa minimikeon taulukon koon tarvittaessa. Koska minimikeon
136      * käyttämän taulukon alkiot manuaalisesti kopioidaan uuteen taulukkoon,
137      * on toiminnon aikavaativuus O(n).
138      */
139     private void resize() {
140         // Luetaan muistiin nykyinen taulukko ja luodaan uusi taulukko.
141         State[] current = this.heap;
142         this.heap = (State[]) new Comparable[this.heap.length * 2];
143         
144         // Manuaalisesti kopioi nykyisen taulukon alkiot uuteen taulukkoon.
145         for (int i = 0; i < this.size; i++) {
146             this.heap[i + 1] = current[i + 1];
147         }
148     }
149 }
